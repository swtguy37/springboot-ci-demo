pipeline {
    agent any

    tools {
        maven 'Maven'   // the Maven tool configured in Jenkins
        jdk 'Java'      // the JDK configured in Jenkins
    }

    stages {
 stage('Clean Workspace') {
            steps {
                deleteDir()
            }
        }
	stage('Parallel Builds'){
parallel {
        stage('Build deployment for Project') {
            steps {
                dir('demo') {  // enter demo folder
                    echo 'Building demo project...'
                   // bat 'mvn clean package'


		bat '''
			netstat -ano ^| findstr: 8082 > pid.txt

			for /f %%a in (pid.txt) do (
			   echo Killing process %%a
 			   taskkill /PID %%a /F
			)
			del pid.txt
			echo Starting application...
			start java -jar target/*.jar
		    '''

                }
            }
        }

        stage('Build Examples Project') {
            steps {
                dir('examples') {  // enter examples folder
                    echo 'Building examples project...'
                    bat 'mvn clean install'
                }
            }
        }

        stage('Build Product Project') {
            steps {
                dir('product') {  // enter product folder
                    echo 'Building product project...'
                    bat 'mvn clean package'
                }
            }
        }

        stage('Final Message') {
            steps {
                echo 'All projects built successfully! ✅'
            }
        }
    }

    post {
        success {
	    archiveArtifacts artifacts: '**/target/*.jar', fingerprint: true 
            echo 'Artifacts Archived ✅'
        }
        failure {
            echo 'Pipeline finished RED ❌'
        }
    }
}

}
}
