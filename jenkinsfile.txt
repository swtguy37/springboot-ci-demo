pipeline {
    agent any

    tools {
        maven 'Maven'   // Maven installed in Jenkins
        jdk 'Java'      // JDK installed in Jenkins
    }

    stages {

       
        stage('Build Projects (Parallel)') {
            parallel {

                stage('Build Demo') {
                    steps {
                        dir('demo') {
                            echo 'Building Demo Project...'
                            bat 'mvn clean package'
                        }
                    }
                }

                stage('Build Examples') {
                    steps {
                        dir('examples') {
                            echo 'Building Examples Project...'
                            bat 'mvn clean install'
                        }
                    }
                }

                stage('Build Product') {
                    steps {
                        dir('product') {
                            echo 'Building Product Project...'
                            bat 'mvn clean package'
                        }
                    }
                }
            }
        }

        stage('Run Demo App on 8082') {
            steps {
                dir('demo') {
                    echo 'Stopping any old Demo on 8082...'
                    bat '''
                    for /f "tokens=5" %%a in ('netstat -ano ^| findstr :8082') do (
                        echo Killing PID %%a
                        taskkill /PID %%a /F
                    )
                    echo Starting Demo App on 8082...
                    java -jar target/demo-0.0.1-SNAPSHOT.jar --server.port=8082                    
                    
                    '''
                }
            }
        }
    }

    post {
        success {
            archiveArtifacts artifacts: '**/target/*.jar', fingerprint: true
            echo 'Artifacts Archived ✅'
        }
        failure {
            echo 'Pipeline finished RED ❌'
        }
    }
}
