pipeline {
    agent any

    tools {
        maven 'Maven'
        jdk 'Java'
    }

    stages {

        stage('Clean Workspace') {
            steps {
                deleteDir()
            }
        }

        stage('Build Projects (Parallel)') {
            parallel {

                stage('Build Demo') {
                    steps {
                        dir('demo') {
                            bat 'mvn clean package'
                        }
                    }
                }

                // Removed Build Examples stage because it has no pom.xml

                stage('Build Product') {
                    steps {
                        dir('product') {
                            bat 'mvn clean package'
                        }
                    }
                }
            }
        }

        stage('Run Demo App on 8082') {
            steps {
                dir('demo') {
                    bat '''
                    netstat -ano | findstr :8082 > pid.txt
                    for /f "tokens=5" %%a in (pid.txt) do taskkill /PID %%a /F
                    del pid.txt

                    echo Starting Demo App on 8082
                    start "" java -jar target/demo-0.0.1-SNAPSHOT.jar --server.port=8082
                    '''
                }
            }
        }
    }

    post {
        success {
            archiveArtifacts artifacts: '**/target/*.jar', fingerprint: true
            echo 'Artifacts Archived ✅'
        }
        failure {
            echo 'Pipeline finished RED ❌'
        }
    }
}
